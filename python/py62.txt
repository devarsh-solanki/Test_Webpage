# list comprehension with if-else statement

numbers = list(range(1,11))
new_list = []
for i in numbers:
    if i%2 == 0:
        new_list.append(i*2)
    else:
        new_list.append(-i)
print(new_list)

new_list__of_numbers = [i*2 if (i%2 == 0) else -i for i in numbers]
print(new_list__of_numbers)  

# nested list

nested_list = [[i for i in range(1,4)] for j in range(3)]
print(nested_list)
