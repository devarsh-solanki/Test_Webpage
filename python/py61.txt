numbers = []
def number_str(l):
    for i in l:
        if (type(i) == int or type(i) == float):
            numbers.append(str(i))
    print(numbers)

print(number_str([1,2,3,1.0,'solanki',1.2,True]))

def number_str2(l):
    return [str(i) for i in l if (type(i) == int or type(i) == float)]

print(number_str2([1,2,3,1.1,True,5,1.0,False,'devarsh']))
